## 1. Project Overview

### Vision
**ChefAI** es un SaaS de generación de recetas culinarias con IA. El usuario introduce los ingredientes que tiene en casa, sus alergias/intolerancias y el número de comensales, y la aplicación genera al instante 3+ recetas completas con imagen, valores nutricionales por ingrediente y por ración, instrucciones paso a paso y ajuste automático de cantidades.

Va dirigido a personas que cocinan en casa, viven con restricciones alimentarias, quieren optimizar lo que tienen en la nevera o simplemente buscan inspiración. Modelo freemium con tres planes (Free, Pro, Chef) y pagos por Stripe y PayPal.

### Goals
- Lanzar un MVP en producción en VPS Hetzner en menos de 2 semanas de desarrollo
- Convertir al menos 3 % de usuarios free a Pro en los primeros 90 días
- Generar al menos 50 recetas por usuario activo mensualmente
- Mantener latencia de generación < 25 s P95 (chat + 1 imagen)

### Success Metrics
- DAU/MAU > 25 %
- Coste OpenAI por usuario activo Pro < 1,50 €/mes (margen sano sobre 9,90 €)
- Conversión landing → registro > 8 %
- Conversión registro → checkout > 5 %

## 4. Data Model

### Prisma Schema Completo

```prisma
// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ---------- AUTH ----------

enum Role {
  USER
  ADMIN
}

model User {
  id              String         @id @default(cuid())
  email           String         @unique
  passwordHash    String
  name            String?
  role            Role           @default(USER)
  emailVerifiedAt DateTime?
  verifyToken     String?        @unique
  resetToken      String?        @unique
  resetTokenExp   DateTime?
  createdAt       DateTime       @default(now())
  updatedAt       DateTime       @updatedAt

  sessions        Session[]
  subscription    Subscription?
  recipes         Recipe[]
  history         RecipeHistory[]
  usage           UsageCounter[]
  payments        Payment[]

  @@index([email])
  @@index([createdAt])
}

model Session {
  id         String   @id @default(cuid())
  userId     String
  tokenHash  String   @unique  // sha256 of JWT jti for revocation
  userAgent  String?
  ip         String?
  expiresAt  DateTime
  createdAt  DateTime @default(now())

  user       User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@index([userId])
  @@index([expiresAt])
}

// ---------- BILLING ----------

enum BillingInterval {
  MONTH
  YEAR
}

enum SubscriptionStatus {
  ACTIVE
  TRIALING
  PAST_DUE
  CANCELED
  INCOMPLETE
}

enum PaymentProvider {
  STRIPE
  PAYPAL
  MANUAL
}

model Plan {
  id                  String          @id @default(cuid())
  slug                String          @unique  // "free", "pro", "chef"
  name                String
  description         String?
  priceCents          Int             // 0 for free
  currency            String          @default("EUR")
  interval            BillingInterval @default(MONTH)
  recipesPerMonth     Int             // -1 = ilimitado
  imagesEnabled       Boolean         @default(false)
  imageQuality        String          @default("low") // low/standard/hd
  pdfExport           Boolean         @default(false)
  shoppingList        Boolean         @default(false)
  weeklyPlanner       Boolean         @default(false)
  prioritySupport     Boolean         @default(false)
  stripePriceId       String?
  paypalPlanId        String?
  isActive            Boolean         @default(true)
  isPublic            Boolean         @default(true)
  sortOrder           Int             @default(0)
  createdAt           DateTime        @default(now())
  updatedAt           DateTime        @updatedAt

  subscriptions       Subscription[]

  @@index([slug])
  @@index([isActive, isPublic])
}

model Subscription {
  id                    String              @id @default(cuid())
  userId                String              @unique
  planId                String
  provider              PaymentProvider
  status                SubscriptionStatus  @default(ACTIVE)
  stripeSubscriptionId  String?             @unique
  stripeCustomerId      String?
  paypalSubscriptionId  String?             @unique
  currentPeriodStart    DateTime
  currentPeriodEnd      DateTime
  cancelAtPeriodEnd     Boolean             @default(false)
  canceledAt            DateTime?
  createdAt             DateTime            @default(now())
  updatedAt             DateTime            @updatedAt

  user                  User                @relation(fields: [userId], references: [id], onDelete: Cascade)
  plan                  Plan                @relation(fields: [planId], references: [id])

  @@index([userId])
  @@index([status])
  @@index([currentPeriodEnd])
}

model Payment {
  id                String           @id @default(cuid())
  userId            String
  provider          PaymentProvider
  externalId        String           // payment_intent / order id
  amountCents       Int
  currency          String           @default("EUR")
  status            String           // succeeded/pending/failed
  rawPayload        Json?
  createdAt         DateTime         @default(now())

  user              User             @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([provider, externalId])
  @@index([userId])
  @@index([createdAt])
}

// ---------- RECIPES ----------

enum RecipeSource {
  AI_GENERATED
  USER_SAVED
}

model Recipe {
  id                String              @id @default(cuid())
  userId            String
  source            RecipeSource        @default(AI_GENERATED)
  title             String
  description       String?
  servings          Int
  prepMinutes       Int?
  cookMinutes       Int?
  difficulty        String?             // easy/medium/hard
  cuisine           String?
  imageUrl          String?
  imageStoragePath  String?             // local /uploads/...
  totalCalories     Float?              // por ración
  totalProteins     Float?              // g por ración
  totalFats         Float?              // g por ración
  totalCarbs        Float?              // g por ración (extra util)
  promptHash        String?             // dedupe / cache
  isFavorite        Boolean             @default(false)
  createdAt         DateTime            @default(now())
  updatedAt         DateTime            @updatedAt

  user              User                @relation(fields: [userId], references: [id], onDelete: Cascade)
  ingredients       RecipeIngredient[]
  steps             RecipeStep[]
  history           RecipeHistory[]

  @@index([userId, createdAt])
  @@index([promptHash])
}

model RecipeIngredient {
  id           String   @id @default(cuid())
  recipeId     String
  name         String
  quantity     Float
  unit         String   // g, ml, ud, cda, ...
  optional     Boolean  @default(false)
  suggested    Boolean  @default(false) // sugerencia añadida por la IA
  calories     Float?
  proteins     Float?
  fats         Float?
  carbs        Float?
  sortOrder    Int      @default(0)

  recipe       Recipe   @relation(fields: [recipeId], references: [id], onDelete: Cascade)

  @@index([recipeId])
}

model RecipeStep {
  id          String   @id @default(cuid())
  recipeId    String
  order       Int
  content     String   @db.Text
  durationMin Int?

  recipe      Recipe   @relation(fields: [recipeId], references: [id], onDelete: Cascade)

  @@unique([recipeId, order])
  @@index([recipeId])
}

model RecipeHistory {
  id           String   @id @default(cuid())
  userId       String
  recipeId     String
  promptInputs Json     // ingredientes, prohibidos, comensales
  tokensUsed   Int?
  costCents    Int?
  createdAt    DateTime @default(now())

  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  recipe       Recipe   @relation(fields: [recipeId], references: [id], onDelete: Cascade)

  @@index([userId, createdAt])
}

// ---------- USAGE / LIMITS ----------

model UsageCounter {
  id          String   @id @default(cuid())
  userId      String
  periodKey   String   // "2026-04" (YYYY-MM)
  recipesUsed Int      @default(0)
  imagesUsed  Int      @default(0)
  tokensUsed  Int      @default(0)
  costCents   Int      @default(0)
  updatedAt   DateTime @updatedAt

  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([userId, periodKey])
  @@index([periodKey])
}

model RateLimit {
  id        String   @id @default(cuid())
  key       String   @unique  // ip:route or userId:action
  count     Int      @default(0)
  windowEnd DateTime
  updatedAt DateTime @updatedAt

  @@index([windowEnd])
}

// ---------- SETTINGS (BRANDING + GLOBAL) ----------

model Settings {
  id           Int      @id @default(1)        // singleton
  brandName    String   @default("ChefAI")
  brandTagline String?
  brandLogoUrl String?
  brandColor   String   @default("#16a34a")    // primary
  supportEmail String   @default("hola@chefai.app")
  termsUrl     String?
  privacyUrl   String?
  updatedAt    DateTime @updatedAt
}
```

### Relationships
- `User 1—N Session` (cascade)
- `User 1—1 Subscription` (cascade)
- `Plan 1—N Subscription`
- `User 1—N Recipe 1—N RecipeIngredient` / `RecipeStep` (cascade)
- `User 1—N RecipeHistory N—1 Recipe`
- `User 1—N UsageCounter` (uno por mes)
- `User 1—N Payment`
- `Settings` es singleton (id=1)

### Seeds (`prisma/seed.ts`)

- 1 fila `Settings` con valores por defecto

### Validation
Toda entrada (server action o API) pasa por un schema Zod (`lib/validators.ts`). Ejemplo:

```ts
export const generateRecipesInput = z.object({
  ingredients: z.array(z.string().trim().min(1)).min(1).max(30),
  forbidden:   z.array(z.string().trim().min(1)).max(30).default([]),
  unwanted:    z.array(z.string().trim().min(1)).max(30).default([]),
  servings:    z.number().int().min(1).max(20),
  cuisine:     z.string().optional(),
  difficulty:  z.enum(["easy","medium","hard"]).optional(),
});
```

### Response shape (server actions)
```ts
type ActionResult<T> =
  | { ok: true; data: T }
  | { ok: false; error: { code: string; message: string; field?: string } };
```

### State Management
- **Server Components por defecto**. Datos vía Prisma directo en el componente.
- **Client Components** solo para interactividad (formularios, modales, toggles).
- **Server Actions** para mutaciones; revalidación con `revalidatePath`/`revalidateTag`.
- **Sin state manager global**. Si surge necesidad, `zustand` (no antes).
- **Caché**: `unstable_cache` con tag por usuario para historial de recetas.

## 8. Authentication & Authorization

### Auth Flow
1. Registro → server action `registerAction`:
   - Valida con Zod (email, password ≥ 10 chars, 1 número, 1 letra)
   - Comprueba que email no exista
   - `bcrypt.hash(password, 12)`
   - Crea `User` (rol USER) + `verifyToken` (cuid)
   - Envía email "Verifica tu cuenta" con link `/verify-email?token=...`
   - Crea sesión y devuelve cookie httpOnly (puede usar la app en modo "no verificado" hasta 24 h)
2. Login → `loginAction`:
   - Valida, busca user, `bcrypt.compare`
   - Si OK: firma JWT con `jose` (HS256, exp 30 días, jti único), guarda hash sha256 del jti en `Session`
   - Setea cookie `session` httpOnly + Secure + SameSite=Lax + Path=/
3. Logout → borra cookie + marca `Session.expiresAt = now()`
4. Forgot password → genera `resetToken` (32 bytes hex), expira en 1 h, envía email
5. Reset password → valida token y expiración, hashea nueva password, invalida todas las sesiones del user

### JWT payload
```ts
type Jwt = { sub: string; jti: string; role: "USER"|"ADMIN"; iat: number; exp: number };
```

### Middleware (`src/middleware.ts`)
- Lee cookie `session`, verifica JWT con `jose`.
- Si falta o inválida en rutas `/dashboard|/admin|/api/recipes|/api/billing` → redirect `/login?next=...`.
- Para `/admin/*` exige `role === "ADMIN"`.
- Aplica rate-limit por IP en rutas `/api/*` y por user en `actions/recipes:generate` (vía wrapper en la action).
- No corre en `/_next`, `/static`, `/api/webhooks/*`, `/api/health`.

### Roles
| Role | Puede |
|------|-------|
| `USER` | Generar/guardar recetas según plan, gestionar suscripción, exportar PDF si plan |
| `ADMIN` | Todo lo anterior + acceder a `/admin/*` (planes, usuarios, branding) |

### Sesiones revocables
JWT por sí solo no revoca. Cada login crea una `Session` con `tokenHash`. `getSession()` además del verify de JWT comprueba que la `Session` siga existiendo y no expirada. Logout = elimina la fila.

```ts
import { NextResponse, type NextRequest } from "next/server";
import { jwtVerify } from "jose";

const SECRET = new TextEncoder().encode(process.env.JWT_SECRET!);
const PROTECTED = [/^\/dashboard/, /^\/generate/, /^\/recipes/, /^\/billing/, /^\/settings/, /^\/admin/];
const ADMIN_ONLY = [/^\/admin/];

export async function middleware(req: NextRequest) {
  const { pathname } = req.nextUrl;
  const needsAuth = PROTECTED.some(r => r.test(pathname));
  if (!needsAuth) return NextResponse.next();

  const token = req.cookies.get("session")?.value;
  if (!token) return NextResponse.redirect(new URL(`/login?next=${pathname}`, req.url));

  try {
    const { payload } = await jwtVerify(token, SECRET);
    if (ADMIN_ONLY.some(r => r.test(pathname)) && payload.role !== "ADMIN") {
      return NextResponse.redirect(new URL("/dashboard", req.url));
    }
    return NextResponse.next();
  } catch {
    return NextResponse.redirect(new URL(`/login?next=${pathname}`, req.url));
  }
}

export const config = {
  matcher: ["/((?!_next/static|_next/image|favicon.ico|api/webhooks|api/health).*)"],
};
```

```ts
export async function rateLimit(key: string, max: number, windowSec: number) {
  const now = new Date();
  const winEnd = new Date(now.getTime() + windowSec * 1000);
  const row = await prisma.rateLimit.upsert({
    where: { key },
    create: { key, count: 1, windowEnd: winEnd },
    update: {
      count: { increment: 1 },
      windowEnd: { set: undefined as any }, // keep window
    },
  });
  if (row.windowEnd < now) {
    await prisma.rateLimit.update({ where: { key }, data: { count: 1, windowEnd: winEnd } });
    return { ok: true, remaining: max - 1 };
  }
  return { ok: row.count <= max, remaining: Math.max(0, max - row.count) };
}
```

```ts
export const openai = new OpenAI({ apiKey: env.OPENAI_API_KEY });

export async function generateRecipes(input: GenerateInput) {
  const { systemPrompt, userPrompt } = buildRecipesPrompt(input);
  const completion = await openai.chat.completions.create({

    ],
```

```markdown
## Architecture

### Directory Structure
- `src/app/` — Rutas (groups: marketing / auth / dashboard / admin) + `api/` (webhooks, health, pdf, og)
- `src/components/` — UI por dominio (`ui/`, `landing/`, `dashboard/`, `admin/`, `auth/`, `shared/`)

- `prisma/` — `schema.prisma`, `seed.ts`, `migrations/`
- `docker/` — Dockerfile, compose, nginx.conf
- `tests/` — `unit/`, `e2e/`

### Data Flow
- UI → **Server Action** → validación Zod → `lib/*` (auth/openai/stripe...) → Prisma → DB
- Webhooks externos → **API Route** → verificación de firma → Prisma
- Server Components leen Prisma directo. Mutaciones SOLO en Server Actions.
- Cache: `unstable_cache` con tags por usuario para historial de recetas; revalidate tras mutaciones.

### Key Patterns
- **Server Components por defecto.** `"use client"` solo donde haya interactividad.
- **Branding dinámico** vía `lib/branding.ts` (lee `Settings` cacheado por request) — usar en `<title>`, navbar, footer, emails.
- **Planes en DB**: nunca hardcodear precios o límites. Usar `lib/plans.ts:getPlan(slug)` y `can(user, "feature")`.
- **Validación en frontera**: cada server action y API route valida con Zod antes de tocar DB.
- **Errores de UI** siempre con shape `{ ok: false, error: { code, message, field? } }` y mensajes en español.
- **Rate limit** envolviendo acciones sensibles con `withRateLimit(key, max, win)`.
- **Logging** con `pino`; nunca `console.log` en código de producción.

## Code Organization Rules

1. **Una entidad por archivo.** Componente, action o helper máx 300 líneas; si excede, extraer.
2. **Path alias** `@/` → `src/`. No imports relativos profundos (`../../..`).
3. **Sin barrel exports.** Importar desde la fuente directa.
4. **Nombres**: archivos `kebab-case`, componentes `PascalCase`, funciones `camelCase`, types `PascalCase`.
5. **TypeScript strict**. Prohibido `any` (usar `unknown` + narrowing). Prohibido `// @ts-ignore`.
6. **Colocación**: componentes específicos de una página → carpeta de la página. Reutilizables → `components/`.
7. **Server Actions** siempre `"use server"` al tope del archivo, jamás dentro de un componente client.
8. **Acceso a DB** únicamente vía `lib/db.ts` (instancia singleton de Prisma).
```

## 17. Prompts de OpenAI (LISTOS PARA USAR)

### `lib/prompts.ts` — Prompt para generación de recetas

**System prompt** (constante, en español para que la IA piense en este idioma):

```
Eres un chef profesional con formación en nutrición. Generas recetas en español, realistas, sabrosas y cocinables en una cocina doméstica estándar (horno, fogones, microondas, batidora). Respondes EXCLUSIVAMENTE con un JSON válido que cumpla el esquema indicado por el usuario, sin texto extra, sin markdown, sin explicaciones.

Reglas estrictas (en orden de prioridad):
1. NUNCA uses ningún ingrediente listado en "forbidden" (alergias/intolerancias). Esto es una cuestión de seguridad alimentaria.
2. EVITA los ingredientes listados en "unwanted" salvo que sean absolutamente imprescindibles; en ese caso, usa una alternativa equivalente.
3. Usa principalmente los ingredientes en "available". Puedes sugerir hasta 3 ingredientes adicionales por receta marcándolos como "suggested": true (sal, aceite, especias comunes no cuentan).
4. Ajusta TODAS las cantidades para el número exacto de "servings" indicado.
5. Los valores nutricionales deben ser realistas y proporcionales (calorías por gramo coherentes con el ingrediente). Calorías y macros se expresan POR RACIÓN para los totales y POR LA CANTIDAD INDICADA para cada ingrediente.
6. Genera al menos 3 recetas DIFERENTES entre sí (técnicas o sabores distintos).
7. Las instrucciones (steps) deben ser claras, en imperativo, numeradas implícitamente por orden, sin omitir pasos críticos (precalentar, salpimentar, reposar).
8. Unidades métricas: g, ml, ud (unidad), cda (cucharada), cdta (cucharadita).
9. Si no es posible generar una receta razonable con las restricciones, devuelve un único objeto en "recipes" con error: { code: "NOT_FEASIBLE", message: "..." } — NO inventes recetas no cocinables.
```

### Validación Zod de la respuesta (`lib/prompts.ts`)

```ts
export const recipesResponseSchema = z.object({
  recipes: z.array(z.object({
    title: z.string().min(2),
    description: z.string(),
    cuisine: z.string().nullable().optional(),
    difficulty: z.enum(["easy","medium","hard"]),
    prepMinutes: z.number().int().min(0),
    cookMinutes: z.number().int().min(0),
    servings: z.number().int().min(1),
    imagePrompt: z.string(),
    ingredients: z.array(z.object({
      name: z.string(),
      quantity: z.number().nonnegative(),
      unit: z.string(),
      optional: z.boolean(),
      suggested: z.boolean(),
      calories: z.number().nonnegative(),
      proteins: z.number().nonnegative(),
      fats: z.number().nonnegative(),
      carbs: z.number().nonnegative(),
    })).min(1),
    perServing: z.object({
      calories: z.number().nonnegative(),
      proteins: z.number().nonnegative(),
      fats: z.number().nonnegative(),
      carbs: z.number().nonnegative(),
    }),
    steps: z.array(z.object({
      order: z.number().int().min(1),
      content: z.string().min(2),
      durationMin: z.number().int().nonnegative().nullable(),
    })).min(2),
  })).min(3),
});
```
